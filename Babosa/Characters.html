def add_approximations(name, hash)
  @approximations ||= {}
  @approximations[name.to_sym] = hash.inject({}) do |memo, object|
    key = object[0].unpack("U").shift
    value = object[1].unpack("C*")
    memo[key] = value.length == 1 ? value[0] : value
    memo
  end
end
